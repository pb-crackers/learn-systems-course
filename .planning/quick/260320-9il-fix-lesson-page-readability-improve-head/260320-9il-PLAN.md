The link should be `text-muted-foreground` and positioned with `absolute -left-6` (the heading gets `relative`).

Implementation directly in mdx-components.tsx (no separate file needed since these are small):

```tsx
function slugify(text: string): string {
  return text.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '')
}

function Heading({ level, children }: { level: 2 | 3; children: React.ReactNode }) {
  const text = typeof children === 'string' ? children : ''
  const slug = slugify(text)
  const Tag = `h${level}` as const
  return (
    <Tag id={slug} className="group relative">
      <a href={`#${slug}`} className="absolute -left-5 top-0 opacity-0 group-hover:opacity-100 transition-opacity text-muted-foreground" aria-hidden="true">#</a>
      {children}
    </Tag>
  )
}
```

Then add to the component map:
```tsx
h2: (props: any) => <Heading level={2} {...props} />,
h3: (props: any) => <Heading level={3} {...props} />,
```

Place these BEFORE the `...components` spread so they can be overridden if needed.

Note: The `children` extraction for slugify needs to handle React nodes. Extract text recursively if children is not a string:
```tsx
function extractText(node: React.ReactNode): string {
  if (typeof node === 'string') return node
  if (typeof node === 'number') return String(node)
  if (Array.isArray(node)) return node.map(extractText).join('')
  if (node && typeof node === 'object' && 'props' in node) return extractText((node as any).props.children)
  return ''
}
```

Then use `extractText(children)` instead of the simple string check.
