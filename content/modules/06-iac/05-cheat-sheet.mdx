        { command: `variable "name" {}`, description: "Declares an input variable with type, description, default, and optional validation block", example: `variable "external_port" {\n  type    = number\n  default = 8080\n}` },

        { command: "tofu init", description: "Downloads required providers into .terraform/ directory. Run once per new directory or after changing required_providers." },
        { command: "tofu fmt", description: "Formats HCL files to canonical style — run before committing to version control" },
        { command: "tofu validate", description: "Validates HCL syntax and provider schema without connecting to infrastructure — fast correctness check" },
        { command: "tofu plan", description: "Shows what changes would be made to infrastructure. Outputs: + create, ~ update, - destroy. Always review before apply." },
        { command: "tofu apply", description: "Applies the planned changes to real infrastructure. Prompts for confirmation unless -auto-approve is passed." },
        { command: "tofu destroy", description: "Destroys all resources tracked in the state file. Always run after exercises to avoid leftover containers." },

## Terraform State

<QuickReference
  sections={[
    {
      title: "State Management (Lesson 3)",
      items: [
        { command: "terraform.tfstate", description: "Local state file written to the working directory after tofu apply. Contains resource IDs, attributes, and metadata. Never edit by hand — use tofu state commands." },
        { command: "tofu state list", description: "Lists all resources currently tracked in state", example: "docker_container.web\ndocker_image.nginx" },
        { command: "tofu state show <resource>", description: "Shows all attributes of a specific resource as stored in state", example: "tofu state show docker_container.web" },
        { command: "tofu state rm <resource>", description: "Removes a resource from state without destroying it in real infrastructure — use when you want OpenTofu to 'forget' a resource" },
        { command: "tofu state mv <old> <new>", description: "Renames a resource in state — use when refactoring HCL resource names without recreating infrastructure" },
        { command: "Remote backends", description: `Store state in a shared location instead of local files. S3: backend "s3" {}. GCS: backend "gcs" {}. Azure: backend "azurerm" {}. Required for team collaboration.` },
        { command: "State locking", description: "Remote backends lock state during operations to prevent concurrent applies from two people or two pipeline runs corrupting the state file." },
        { command: ".gitignore terraform.tfstate", description: "Never commit state files to git — they contain sensitive resource attributes (IP addresses, IDs, sometimes secrets). Add terraform.tfstate and terraform.tfstate.backup to .gitignore." },
      ]
    }
  ]}

## Modules

<QuickReference
  sections={[
    {
      title: "Module Patterns (Lesson 4)",
      items: [
        { command: `module "name" { source = "..." }`, description: "Calls a child module and passes input variables. The module block is in your root module; the referenced directory is the child module.", example: `module "frontend" {\n  source        = "./modules/web-container"\n  name          = "learn-frontend"\n  external_port = 8081\n}` },
        { command: "Root module vs child module", description: "Root module: the directory where you run tofu init/plan/apply. Child module: any module called via a module block. Root can call many children; children can call other children." },
        { command: `Local module source`, description: `source = "./modules/name" — relative path from the root module to the child module directory. tofu init copies the module into .terraform/modules/.`, example: `source = "./modules/web-container"` },
        { command: "Registry module source", description: `source = "registry.opentofu.org/namespace/module/provider" — downloads published module from registry. Used for community modules like VPC configurations.`, example: `source  = "registry.opentofu.org/kreuzwerker/docker"\nversion = "3.9.0"` },
        { command: "Module inputs (variables.tf)", description: "A child module's variables.tf defines its interface — the inputs the caller must (or can optionally) pass. Think of it as a function signature." },
        { command: "Module outputs (outputs.tf)", description: "A child module's outputs.tf defines what values it returns to the caller. Access via module.name.output_name in the root module.", example: "# In root module:\nmodule.frontend.container_id\nmodule.api.container_name" },
        { command: "tofu init initializes modules", description: "Running tofu init after adding or changing module sources downloads/updates the referenced modules into .terraform/modules/." },
      ]
    }
  ]}
